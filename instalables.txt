            public function toggleVisibility(BloodType $bloodType)
    {
        $companyCode = auth()->user()->code_company;
        $companyView = $bloodType->company_view ?? [];

        if (in_array($companyCode, $companyView)) {
            $companyView = array_values(array_filter($companyView, fn($c) => $c !== $companyCode));
        } else {
            $companyView[] = $companyCode;
        }

        $bloodType->update(['company_view' => $companyView]);
        return redirect()->back();
    }
